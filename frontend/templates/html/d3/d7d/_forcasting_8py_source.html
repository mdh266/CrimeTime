137  def stationarity(self, timeseries):
138  """
139  Perform Dickey-Fuller test:
140  """
141  # used from
142  #http://www.seanabu.com/2016/03/22/time-series-seasonal-ARIMA-model-in-python/
143  #Determing rolling statistics
144  rolmean = timeseries.rolling(window=12,center=False).mean()
145  rolstd = timeseries.rolling(window=12,center=False).std()
146 
147  #Plot rolling statistics:
148  fig = plt.figure(figsize=(8, 6))
149  orig = plt.plot(timeseries, color='blue',label='Original')
150  mean = plt.plot(rolmean, color='red', label='Rolling Mean')
151  std = plt.plot(rolstd, color='black', label = 'Rolling Std')
152  plt.legend(loc='best')
153  plt.title('Rolling Mean & Standard Deviation')
154  plt.show()
155 
156  print 'Results of Dickey-Fuller Test:'
157  dftest = adfuller(timeseries, autolag='AIC')
158  dfoutput = pd.Series(dftest[0:4], index=['Test Statistic','p-value',
159  '#Lags Used','Number of Observations Used'])
160  for key,value in dftest[4].items():
161  dfoutput['Critical Value (%s)'%key] = value
162  print dfoutput
